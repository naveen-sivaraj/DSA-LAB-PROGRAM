void merge(int a[],int low ,int mid,int high){
int i,j,k;
int b[50];
i=low;
j=mid+1;
k=low;
while(i<=mid&&j<=high){
if(a[i]<=a[j]){
b[k]=a[i];
i++;
}else{
b[k]=a[j];
j++;
}
k++;
}
while(i<=mid){
b[k]=a[i];
i++;
k++;
}
while(j<=high){
b[k]=a[j];
j++;
k++;
}
for(i=low;i<=high;i++){
a[i]=b[i];
}
}
void mergeSort(int a[],int low,int high){
int mid;
if(low<high){
mid = (low +high)/2;
mergeSort(a,low,mid);
mergeSort(a,mid+1,high);
merge(a,low,mid,high);
}
}
