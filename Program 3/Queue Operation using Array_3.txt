#include <stdio.h>
#include <stdlib.h>

#define MAX 5
int queue[MAX];
int front = -1;
int rear = -1;

void insert(int x) {
    if (rear == MAX - 1) {
        printf("\nQueue Full\n");
        return;
    }
    rear++;
    queue[rear] = x;
    if (front == -1)
        front = 0;
}

int delete() {
    int val;
    if (front == -1) {
        printf("\nQueue Empty\n");
        return -1;
    }
    val = queue[front];
    if (front == rear)
        front = rear = -1;
    else
        front++;
    return val;
}

void view() {
    if (front == -1) {
        printf("\nQueue Empty\n");
    } else {
        printf("\nFront --> ");
        for (int i = front; i <= rear; i++)
            printf("%4d", queue[i]);
        printf(" <-- Rear\n");
    }
}

int main() {
    int ch = 0, val;

    while (1) {
        printf("\nQUEUE OPERATION\n");
        printf("1. INSERT\n2. DELETE\n3. VIEW\n4. QUIT\n");
        printf("Enter Choice: ");
        scanf("%d", &ch);

        switch (ch) {
            case 1:
                printf("Enter element to be inserted: ");
                scanf("%d", &val);
                insert(val);
                break;

            case 2:
                val = delete();
                if (val != -1)
                    printf("Element deleted: %d\n", val);
                break;

            case 3:
                view();
                break;

            case 4:
                printf("Exiting...\n");
                exit(0);

            default:
                printf("Invalid Choice\n");
        }
    }

    return 0;
}
