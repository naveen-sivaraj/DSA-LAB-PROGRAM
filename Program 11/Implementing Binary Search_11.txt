#include <stdio.h>
#include <conio.h>
void main()
{
 int n,key,mid,high,i,found,low;
 int arr[100];
 clrscr();
 printf("Enter the number of elements: ");
 scanf("%d",&n);
 printf("Enter %d sorted elements :\n",n);
 for(i=0;i<n;i++)
 {
  scanf("%d",&n);
 }
 printf("Enter the element to search: ");
 scanf("%d",&key);
 low =0;
 high=n-1;
 found=0;
 while(low<=0)
 {
  mid=(low+high)/2;
  if(arr[mid]==key)
  {
  printf("Element found at position %d (index %d).\n",mid+1,mid);
  found=1;
  break;
  }
  else if(arr[mid]<key)
  {
  low=mid+1;
  }
  else {
  high =mid+1;
  }
}
if (found)
{
  printf("Element not found in the array.\n");
  }
  getch();
}
